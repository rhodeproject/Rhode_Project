def as_json(options = {})
  {
      :id => self.id,
      :title => self.title,
      :description => self.description || "",
      :start => starts_at,
      :end => ends_at,
      :allDay => self.all_day,
      :recurring => false,
      :url => Rails.application.routes.url_helpers.event_path(id)
  }
