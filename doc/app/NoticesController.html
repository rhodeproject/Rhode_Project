def create
  club = Club.find(current_user.club.id)
  @notice = club.notices.build(params[:notice])
  if @notice.save
    @notice.send_tweet(club)
    flash[:success] = "Notice Added!"
  else
    flash[:success] = "Failed to add Notice"
  end
  redirect_to notices_path
end
